# ✅ 2. Round Robin Scheduling

### Program

```c
#include<stdio.h>

int main(){
  int n,tq,time=0,rem[10],bt[10];

  printf("Enter n & tq: ");
  scanf("%d%d",&n,&tq);

  for(int i=0;i<n;i++){
    scanf("%d",&bt[i]);
    rem[i]=bt[i];
  }

  while(1){
    int done=1;
    for(int i=0;i<n;i++){
      if(rem[i]>0){
        done=0;
        if(rem[i]>tq){
          time+=tq;
          rem[i]-=tq;
        } else{
          time+=rem[i];
          rem[i]=0;
        }
      }
    }
    if(done) break;
  }

  printf("Total Time=%d",time);
}
```

### Sample Output

```
Enter n & tq:
3 2
5 3 1
Total Time=9
```

# ✅ 4. FIFO Page Replacement

### Program

```c
#include<stdio.h>

int main(){
  int f=3,p[10],n,frame[3],k=0,fault=0;

  scanf("%d",&n);
  for(int i=0;i<n;i++) scanf("%d",&p[i]);

  for(int i=0;i<f;i++) frame[i]=-1;

  for(int i=0;i<n;i++){
    int found=0;
    for(int j=0;j<f;j++)
      if(frame[j]==p[i]) found=1;

    if(!found){
      frame[k]=p[i];
      k=(k+1)%f;
      fault++;
    }
  }

  printf("Faults=%d",fault);
}
```

### Sample Output

```
5
1 2 3 1 4
Faults=4
```

# ✅ 8. Banker’s Algorithm

### Program

```c
#include<stdio.h>

int main(){
  int alloc[2][2]={{1,0},{0,1}};
  int max[2][2]={{2,1},{1,2}};
  int need[2][2];

  for(int i=0;i<2;i++)
    for(int j=0;j<2;j++)
      need[i][j]=max[i][j]-alloc[i][j];

  printf("Need Matrix:\n");

  for(int i=0;i<2;i++){
    for(int j=0;j<2;j++)

  }
}
```

# ✅ 10. Producer–Consumer

### Program

```c
#include<stdio.h>
#include<pthread.h>
#include<semaphore.h>

sem_t empty,full;

void* producer(){
  sem_wait(&empty);
  printf("Produced\n");
  sem_post(&full);
}

void* consumer(){
  sem_wait(&full);
  printf("Consumed\n");
  sem_post(&empty);
}

int main(){
  pthread_t t1,t2;

  sem_init(&empty,0,1);
  sem_init(&full,0,0);

  pthread_create(&t1,NULL,producer,NULL);
  pthread_create(&t2,NULL,consumer,NULL);

  pthread_join(t1,NULL);
  pthread_join(t2,NULL);
}
```

### Sample Output

```
Produced
Consumed
```

# ✅ 11. SJF Scheduling

```c
#include<stdio.h>

int main(){
  int n,bt[10];

  scanf("%d",&n);
  for(int i=0;i<n;i++) scanf("%d",&bt[i]);

  for(int i=0;i<n;i++)
    for(int j=i+1;j<n;j++)

        int t=bt[i];
        bt[i]=bt[j];
        bt[j]=t;

}
```

# ✅ 15. SSTF Disk Scheduling

### Program

```c
#include<stdio.h>
#include<stdlib.h>

int main(){
  int req[5]={98,183,37,122,14},head=53;

  for(int i=0;i<5;i++){
    int min=999,idx;

    for(int j=0;j<5;j++){
      int d=abs(head-req[j]);
      if(d<min){
        min=d;
        idx=j;
      }
    }

    printf("%d -> ",req[idx]);
    head=req[idx];
    req[idx]=999;
  }
}
```

### Sample Output

```
43 -> 24 -> 14 -> 82 -> 122 ->
```
