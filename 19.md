# 💻 **Java Simulation Program**

👉 This simulates:

- Packet transmission
    
- Delay
    
- Packet loss
    
- Throughput
    

```java
import java.util.Random;

public class RoutingPerformance {

    static void simulateRIP(int packets) {
        Random rand = new Random();
        int delivered = 0;
        int delay = 0;

        for (int i = 0; i < packets; i++) {
            delay += rand.nextInt(15) + 10; // higher delay

            // packet loss chance
            if (rand.nextDouble() > 0.2) {
                delivered++;
            }
        }

        System.out.println("\n--- RIP Performance ---");
        System.out.println("Packets Sent: " + packets);
        System.out.println("Packets Delivered: " + delivered);
        System.out.println("Total Delay: " + delay);
        System.out.println("Throughput: " + (delivered * 1000 / delay));
    }

    static void simulateOSPF(int packets) {
        Random rand = new Random();
        int delivered = 0;
        int delay = 0;

        for (int i = 0; i < packets; i++) {
            delay += rand.nextInt(8) + 5; // lower delay

            if (rand.nextDouble() > 0.1) {
                delivered++;
            }
        }

        System.out.println("\n--- OSPF Performance ---");
        System.out.println("Packets Sent: " + packets);
        System.out.println("Packets Delivered: " + delivered);
        System.out.println("Total Delay: " + delay);
        System.out.println("Throughput: " + (delivered * 1000 / delay));
    }

    public static void main(String[] args) {
        int packets = 50;

        simulateRIP(packets);
        simulateOSPF(packets);
    }
}
```
