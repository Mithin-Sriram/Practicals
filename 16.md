# 💻 **Java Simulation Program**

👉 This simulates:

- Data transmission
    
- Delay
    
- Packet loss
    
- Throughput
    

```java
import java.util.Random;

public class TCP_UDP_Simulation {

    static void simulateTCP(int packets) {
        Random rand = new Random();
        int delivered = 0;
        int delay = 0;

        for (int i = 1; i <= packets; i++) {
            delay += rand.nextInt(10) + 5; // higher delay

            // TCP retransmits lost packets
            if (rand.nextDouble() < 0.2) {
                delay += 10; // retransmission delay
            }

            delivered++;
        }

        System.out.println("\n--- TCP Simulation ---");
        System.out.println("Packets Sent: " + packets);
        System.out.println("Packets Delivered: " + delivered);
        System.out.println("Total Delay: " + delay);
        System.out.println("Throughput: " + (delivered * 1000 / delay));
    }

    static void simulateUDP(int packets) {
        Random rand = new Random();
        int delivered = 0;
        int delay = 0;

        for (int i = 1; i <= packets; i++) {
            delay += rand.nextInt(5) + 1; // low delay

            // UDP packet loss (no retransmission)
            if (rand.nextDouble() > 0.2) {
                delivered++;
            }
        }

        System.out.println("\n--- UDP Simulation ---");
        System.out.println("Packets Sent: " + packets);
        System.out.println("Packets Delivered: " + delivered);
        System.out.println("Total Delay: " + delay);
        System.out.println("Throughput: " + (delivered * 1000 / delay));
    }

    public static void main(String[] args) {
        int packets = 50;

        simulateTCP(packets);
        simulateUDP(packets);
    }
}
```
